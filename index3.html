<style>
#slider3 figcaption {
    display: block;
    position: absolute;
    bottom: 0;
    left: 0;
    right: 0;
    margin: 0;

Set the fade duration through the stylesheet as classic transitions.

JS

<script>
document.addEventListener('DOMContentLoaded', function() {
  // Slideshow 3
  new responsiveSlides( '#slider3' );
});
</script>
